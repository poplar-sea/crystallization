# 1.绘制直线
```js
// 1. 获取canvasdom节点
var cnv = document.getElementById("canvas");
// 2.获取canvas上下文对象
var cxt = cnv.getContext("2d")
// 3.moveTo()的含义是“将画笔移到该点的位置，然后开始绘图”。
cxt.moveTo(50,100)
// 4.lineTo()的含义是“将画笔从起点开始画直线，一直到终点”。
cxt.lineTo(150, 250)
cxt.lineTo(150, 100)
// 5.绘制
cxt.stroke()
```
# 2.绘制描边矩形
```js
var cnv = document.getElementById("canvas");
var cxt = cnv.getContext("2d")
/**
 * strokeStyle属性取值有3种：颜色值、渐变色和图案。
 */
// 1.颜色值
cxt.strokeStyle = "red";
/**
 * strokeRect(x,y,width,height)方法用于确定矩形的坐标;
 * (x,y)为矩形左上角点的坐标
 * width表示矩形的宽度， height表示矩形的高度，
 * 默认情况下width和height都是以px为单位的
 */
cxt.strokeRect(200, 200, 80, 80);
// 上述代码等价于：
/**
 * rect(x,y,width,height)方法用于确定矩形的坐标;
 * (x,y)为矩形左上角点的坐标
 * width表示矩形的宽度， height表示矩形的高度，
 * 默认情况下width和height都是以px为单位的
 */
cxt.rect(200, 200, 80, 80);
cxt.stroke();
```
# 3.绘制填充矩形
```js
var cnv = document.getElementById("canvas");
var cxt = cnv.getContext("2d")
/**
 * fillStyle属性取值有3种：颜色值、渐变色和图案。
 */
// 1.颜色值
 cxt.fillStyle = "red";
/**
 * fillRect(x,y,width,height)方法用于确定矩形的坐标;
 * (x,y)为矩形左上角点的坐标
 * width表示矩形的宽度， height表示矩形的高度，
 * 默认情况下width和height都是以px为单位的
 */
cxt.fillRect(200, 200, 80, 80);

// 上述代码等价于：
cxt.rect(50,50,80,80);
cxt.fill();
/**
 * clearRect(x,y,width,height)方法用于清空“指定矩形区域”;
 * (x,y)为矩形左上角点的坐标
 * width表示矩形的宽度， height表示矩形的高度，
 * 默认情况下width和height都是以px为单位的
 */
cxt.clearRect(60, 60, 50, 50);
```
